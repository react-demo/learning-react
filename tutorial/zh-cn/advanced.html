tests

使用 mocha/expect.js 编写 spec js 文件，可使用 React.addons.TestUtils 套件 


import React from 'react';
import Menu from '@alipay/biz-menu';
import '@alipay/biz-menu/assets/index.less';
var TestUtils = React.addons.TestUtils;
describe('@alipay/biz-menu', () => {
    it('works', () => {
        var instance = TestUtils.renderIntoDocument(React.createElement(Menu));
        expect(TestUtils.scryRenderedDOMComponentsWithClass('.rc-menu-item')).to.be.ok();
    });
});

如何开发一个项目

  • 纯前端

纯前端项目


                    antd init

package.json


{
    "entry": {
        "index": "./index.js"
    },
    "dependencies": {
        "antd": "*"
    },
    "devDependencies": {
        "antd-bin": "0.x"
    },
    "scripts": {
        "dev": "antd server",
        "build": "antd build"
    }
}

entry: 入口文件集合

一个应用页面 index.html

安装依赖


                    tnpm install react antd --save

入口文件

index.js: 引用依赖，渲染 react 组件到页面空位 


import React from 'react';
import 'antd/lib/index.css';
import Datepicker from 'antd/lib/datepicker';

class App {
    render(){
        return <Datepicker />;
    }
}

React.render(<App />, document.getElementById('react-content'));
